function updateHubUI() { const depts = [ { id: 'police', name: 'Law Enforcement', url: 'police.html' }, { id: 'ems', name: 'Medical Service', url: 'fire.html' }, { id: 'dot', name: 'Department of Transport', url: 'dot.html' }, { id: 'civ', name: 'Civilian Portal', url: 'civ.html' } ]; depts.forEach(dept => { const card = document.getElementById(dept.id + '-card'); const statusDiv = document.getElementById(dept.id + '-status'); const btn = document.getElementById(dept.id + '-btn'); const isLocked = localStorage.getItem(dept.id + "_locked") !== "false"; if (isLocked) { card.classList.add('disabled'); statusDiv.innerText = "● Maintenance In Progress"; statusDiv.style.color = "#ffbc00"; btn.style.display = "none"; } else { card.classList.remove('disabled'); statusDiv.innerText = "● System Online"; statusDiv.style.color = "#4ade80"; btn.style.display = "block"; btn.onclick = () => { // Save context so login.html knows where we are going localStorage.setItem("selected_dept_id", dept.id); localStorage.setItem("selected_dept_name", dept.name); localStorage.setItem("selected_dept_url", dept.url); window.location.href = 'login.html'; }; } }); } window.onload = updateHubUI;
